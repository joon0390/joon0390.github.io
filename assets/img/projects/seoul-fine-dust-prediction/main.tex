\begin{frame}{데이터 오류 처리: 보간 (Interpolation)}
    \begin{block}{\small 처리 과정}
        \begin{enumerate}\scriptsize
            \item\textbf{오류 탐지}:
            물리적으로 불가능한 음수인(-) 농도 값을 탐지
            \vspace{0.5cm}
            \item\textbf{결측치 변환}:
            탐지된 오류 값을 비어있는 값(NaN)으로 변환
            \vspace{0.5cm}
            \item\textbf{Cubic 보간}:
            주변의 정상 값들을 이용하여 3차 곡선을 생성하고, 비어있는 지점의 값을 추정하여 삽입
            \vspace{0.5cm}
            \item\textbf{최종 보정}:
            보간 과정에서 발생할 수 있는 음수 값을 0으로 최종 처리하여 보간의 목적을 달성
        \end{enumerate}
    \end{block}
\end{frame}

                    \vspace{0.3cm}
                    \item \textbf{문제점}: 이러한 분포는 일부 극단적으로 높은 값이 모델에 과도한 영향을 미침

                \end{itemize}

                \label{fig:log_dist}

        % --- 오른쪽 열 (텍스트 설명) ---
        \begin{column}{0.4\textwidth}
            \begin{block}{\small 로그 변환 효과}
                \begin{itemize}\scriptsize
                    \item\textbf{대칭성 증가}: \\
                    오른쪽으로 길게 치우쳤던 분포가 중앙으로 모여 \textbf{정규분포와 유사한} 대칭적인 형태로 변환됨
                    \vspace{0.2cm}
                    \item\textbf{이상치 영향 완화}: \\
                    극단적으로 컸던 값들이 다른 값들과의 격차가 줄어들어, 모델 학습 시 \textbf{이상치의 과도한 영향력을 줄일 수 있음.}
                    \vspace{0.2cm}
                    \item\textbf{안정화된 타겟변수}: \\
                    "로그 변환된 타겟 변수"를 새로운 컬럼으로 추가하여, 분산이 안정화된 새로운 타겟변수 생성.
                \end{itemize}
            \end{block}
        \end{column}
    \end{columns}
\end{frame}
%-----------------------------------------------------------------------------------
\begin{frame}{타겟 변수의 자기상관 분석 (로그 변환 후)}
    
    \begin{columns}[T]
        \begin{column}{0.6\textwidth}
            \begin{figure}
                \centering
                \includegraphics[width=\linewidth]{eda/acf.png}
                \caption{\scriptsize ACF \& PACF Plots}
                \label{fig:autocorrelation}
            \end{figure}
        \end{column}

        \begin{column}{0.4\textwidth}
            \scriptsize
            \begin{itemize}
                \item \textbf{ACF}:\\
                상관계수가 천천히 감소하는 것은 데이터에 강한 \textbf{추세(Trend)가 존재}하며, 비정상성(Non-stationary)임을 의미
                \vspace{0.3cm}
                \item \textbf{PACF}:\\
                Lag 1 이후 상관계수가 급격히 감소하는 것은 현재 데이터가 \textbf{바로 직전(1일 전) 데이터와 가장 강한 직접적 관련}이 있음을 시사. (AR 특성)
            \end{itemize}
        \vspace{0.3cm}
        \begin{itemize}
            \item \scriptsize ACF의 패턴을 통해 데이터의 추세를 제거하고 정상성(Stationary)을 확보하고자 \textbf{1차 차분(d=1)을 적용}
        \end{itemize}
        \end{column}
    \end{columns}
\end{frame}
%-----------------------------------------------------------------------------------
\begin{frame}{1차 차분 후 자기상관 분석}

    \begin{columns}[T]
        \begin{column}{0.6\textwidth}
            \begin{figure}
                \includegraphics[width=\linewidth]{eda/acf_after.png}
                \caption{\scriptsize 1차 차분을 적용한 데이터의 ACF \& PACF}
                \label{fig:diff_acf_pacf}
            \end{figure}
        \end{column}

        \begin{column}{0.4\textwidth}
            \scriptsize
            \begin{itemize}
                \item 1차 차분을 통해 데이터의 \textbf{추세가 제거}되고 안정성이 확보됨.
                \vspace{0.3cm}
                \item \textbf{ACF}가 특정 시점 이후 절단되어 \textbf{MA(q)항}의 차수를 결정할 수 있음 (q$:=$1)
                \vspace{0.3cm}
                \item \textbf{PACF} 역시 특정 시점 이후 절단되어 \textbf{AR(p)항}의 차수를 결정할 수 있음. (p$:=$1)
            \end{itemize}
            \vspace{0.5cm}
            \begin{itemize}
                \item  \scriptsize ACF와 PACF가 모두 절단되어 \textbf{ARIMA(2, 1, 2)} 모델이 적합함을 의미
            \end{itemize}
        \end{column}
        
    \end{columns}
\end{frame}

        \end{itemize}
    \end{block}

\begin{frame}{공간적 자기상관성 분석: Moran's I}
    \begin{block}{\small Global Moran's I}
        $$
        I = \frac{N}{\sum_{i=1}^{N}\sum_{j=1}^{N} w_{ij}} \frac{\sum_{i=1}^{N}\sum_{j=1}^{N} w_{ij}(x_i - \bar{x})(x_j - \bar{x})}{\sum_{i=1}^{N}(x_i - \bar{x})^2}
        $$
    \end{block}
        
    \begin{columns}[T]
        % --- 왼쪽 열 ---
        \begin{column}{0.5\textwidth}
            \begin{itemize}\scriptsize
                \item $N$: 총 측정소의 개수
                \item $x_i$: 측정소 $i$의 값
                \item $\bar{x}$: 전체 측정소의 평균 값
            \end{itemize}
        \end{column}
        % --- 오른쪽 열 ---
        \begin{column}{0.5\textwidth}
            \begin{itemize}\scriptsize
                \item $w_{ij}$: 측정소 $i$와 $j$ 간의 공간적 관계를 나타내는 \textbf{공간 가중치}. (ex. if adjacent 1 else 0)
            \end{itemize}
        \end{column}
    \end{columns}
    
    \vspace{0.3cm}
    
    \begin{itemize}
        \item \scriptsize Moran's I는 기본적으로 \textbf{인접한 지역 간의 값이 서로 유사한지}를 측정함. $(x_i - \bar{x})(x_j - \bar{x})$ 부분은 두 지역의 값이 전체 평균보다 둘 다 높거나 둘 다 낮은지를 확인하며, 이 값을 공간 가중치($w_{ij}$)로 조절하여 전체적인 군집 패턴을 계산함
        \item PM2.5\_log의 평균농도 값은 \textbf{0.2320(p-val : 0.0120)}, PM10\_log의 값은 \textbf{0.0290(p-val : 0.2270)}으로 PM2.5\_log가 통계적으로 유의미한 약한 양의 자기상관이 존재
    \end{itemize}    
\end{frame}

\begin{frame}{PM2.5 공간 자기상관성 분석 (Moran Scatter Plot)}
    \begin{columns}[T]
        % --- 왼쪽: 시각화 ---
        \begin{column}{0.6\textwidth}
            \begin{figure}
                \centering
                \includegraphics[width=\linewidth]{eda/change_of_moransI.png}
                \caption{\scriptsize 전체 기간 평균 PM2.5 농도에 대한 산점도}
            \end{figure}
        \end{column}
        
        % --- 오른쪽: 해석 ---
        \begin{column}{0.4\textwidth}
            \begin{itemize}\scriptsize
                \item Moran's I 통계량이 겨울과 봄에 높아지며, 여름과 가을에 낮아지는 패턴
                \vspace{0.3cm}
                \item 이는 미세먼지 농도의 \textbf{공간적 분포 특성}이 고정되어 있지 않고, 계절별 \textbf{기상 조건}에 따라 동적으로 변화함을 의미
                \vspace{0.3cm}
                \item \textbf{겨울,봄철}: 대기 정체로 인해 오염물질이 발생 지역 주변에 머물며 \textbf{강한 공간 군집} 형성 $\rightarrow$ 높은 Moran's I
                \vspace{0.3cm}
                \item \textbf{여름,가을철}: 활발한 대기 순환 및 강수로 오염물질이 도시 전역으로 희석되어 \textbf{무작위 패턴}에 가까워짐 $\rightarrow$ 낮은 Moran's I
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{시공간 파생변수 추가}
    \begin{block}{\small EDA 결과를 변수로 변환}
        \scriptsize
        EDA에서 발견한 미세먼지의 동적인 시공간 패턴을 모델이 학습할 수 있도록, '공간 지연(Spatial Lag)' 변수와 '계절 상호작용(Seasonal Interaction)' 변수를 생성
    \end{block}
    \vspace{0.5cm}
    \begin{columns}[T]
        \begin{column}{0.5\textwidth}
            \begin{block}{\scriptsize \textbf{1. 공간 지연(Spatial Lag) 변수}}
                \begin{itemize}\scriptsize
                    \item 각 측정소의 특정 시점에서, 사전에 정의된 '이웃' 측정소들의 평균 농도를 계산한 값
                    \item "주변 관측소에서의 미세먼지 농도는 현재 이렇다"라는 가장 직접적인 공간적 맥락
                    \item \textbf{생성 변수}: \texttt{PM10\_spatial\_lag}, \texttt{PM2.5\_spatial\_lag}
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{block}{\scriptsize \textbf{2. 계절 상호작용(Seasonal Interaction) 변수}}
                \begin{itemize}\scriptsize
                    \item '공간 지연' 변수를 계절에 따라 두 개의 변수로 분리한 것
                    \item 공간적 군집이 겨울, 봄에 강해지는 EDA 결과를 모델에 반영하였음. 이를 통해 모델은 계절에 따라 공간 정보의 중요도를 다르게 학습하게 됨.
                    \item \textbf{생성 변수}: \texttt{PM10\_sl\_ws}, \texttt{PM10\_sl\_sf}, \texttt{PM2.5\_sl\_ws}, \texttt{PM2.5\_sl\_sf}
                \end{itemize}
            \end{block}
        \end{column}
    \end{columns}
\end{frame}
